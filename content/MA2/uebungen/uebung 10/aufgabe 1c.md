$$
f(t) = -\frac{1}{250}t^3 + \frac{1}{10}t^2
$$

# 3 Definitionsbereich

$$

$$

# 4 Ableiten

$$
f'(t) = -\frac{3}{250}t^2 + \frac{2}{10}t
$$

# 5 Nullstellen

$$
\begin{align*}
	f'(t) &= 0 \\
	-\frac{3}{250} t^2 + \frac{2}{10} t &= 0 \\
	\frac{2}{10}t &= \frac{3}{250}t^2 \\
	\frac{2}{10} &= \frac{3}{250}t \\
	\frac{2 \cdot 250}{10 \cdot 3} &= t \\
	t &= \frac{50}{3}
\end{align*}
$$
